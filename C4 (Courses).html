Pointers and Arrays in C

Introduction to Pointers

A pointer is a variable that stores the address of another variable.

#include <stdio.h> int main() { int a = 10; int *ptr = &a; // Pointer declaration printf("Value of a = %d\n", a); printf("Address of a = %p\n", &a); printf("Pointer value (ptr) = %p\n", ptr); printf("Value at pointer address = %d\n", *ptr); return 0; }

Pointer Arithmetic

Pointer arithmetic allows performing operations like increment or decrement directly on pointer variables.

#include <stdio.h> int main() { int arr[3] = {10, 20, 30}; int *ptr = arr; printf("Element 1: %d\n", *ptr); ptr++; printf("Element 2: %d\n", *ptr); ptr++; printf("Element 3: %d\n", *ptr); return 0; }
Try It Now

Pointer to Arrays

A pointer can be used to access and manipulate array elements efficiently.

#include <stdio.h> int main() { int arr[5] = {1, 2, 3, 4, 5}; int *ptr = arr; printf("Array Elements: "); for (int i = 0; i < 5; i++) { printf("%d ", *(ptr + i)); } return 0; }

Pointer to Pointer (Double Pointer)

A pointer to pointer stores the address of another pointer.

#include <stdio.h> int main() { int a = 10; int *ptr1 = &a; // Pointer to integer int **ptr2 = &ptr1; // Pointer to pointer printf("Value of a = %d\n", a); printf("Value using ptr1 = %d\n", *ptr1); printf("Value using ptr2 = %d\n", **ptr2); return 0; }

Passing Pointers to Functions

Passing a pointer to a function allows direct modification of variable values.

#include <stdio.h> void modifyValue(int *ptr) { *ptr = 50; } int main() { int num = 10; printf("Before modification: %d\n", num); modifyValue(&num); printf("After modification: %d\n", num); return 0; }

Pointer with Strings

Pointers simplify string handling by directly pointing to character arrays.

#include <stdio.h> int main() { char str[] = "CODENEXUS"; char *ptr = str; printf("String using pointer: "); while (*ptr != '\0') { printf("%c", *ptr); ptr++; } return 0; }
